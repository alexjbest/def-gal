so

\begin{equation*} \phi (gh) = \phi (g) \bar \rho (h)+\bar \rho (g) \phi (h) \end{equation*}

\begin{equation*} \bar \rho (h)\inv \bar \rho (g)\inv = \bar \rho (gh)\inv \text{.} \end{equation*}
\begin{equation*} \phi (gh) \bar \rho (gh)\inv = \phi (g) \bar \rho (g)\inv +\bar \rho (g) \phi (h) \bar \rho (h)\inv \bar \rho (g)\inv \end{equation*}

\begin{equation*} \psi (g) = \phi (g) \bar \rho (g)\inv \end{equation*}

\begin{equation*} \psi (gh) = \psi (g) + \bar \rho ( g) \psi (h) \bar \rho (g)\inv \end{equation*}

\begin{equation*} g \bullet M = \bar \rho (g) M \bar \rho (g)\inv \text{.} \end{equation*}
