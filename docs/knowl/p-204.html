\begin{equation*} \bar\rho \colon G_{K,S} \to \GL_2(k) \end{equation*}
\begin{equation*} S= \{ p, \infty \} \end{equation*}
\begin{equation*} \krulldim(R_{\bar\rho }/p R_{\bar \rho }) = 4 + 1 - \dim H^0(G_{\RR}, \ad \bar \rho ) \end{equation*}

take

\begin{equation*} c \in \Gal \CC \RR \end{equation*}

either

\begin{equation*} \bar \rho (c) =\pm \begin{pmatrix} 1 \amp 0 \\ 0 \amp 1\end{pmatrix} \end{equation*}

or

\begin{equation*} \bar \rho (c) \sim \begin{pmatrix} 1 \amp 0 \\ 0 \amp -1\end{pmatrix} \end{equation*}

the first case is called even the second odd. In the even case we have

\begin{equation*} \ad \bar \rho = \mathbf 1 \end{equation*}

and so \(H^0\) has dimension 4. And

\begin{equation*} \krulldim(R_{\bar\rho }/p R_{\bar \rho }) = 1 \end{equation*}
\begin{equation*} \dim(R_{\bar\rho }/p R_{\bar \rho }) \ge 1\text{.} \end{equation*}
